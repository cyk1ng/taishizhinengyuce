function showWeatherModal() {
    // 创建弹窗
    const modal = document.createElement('div');
    modal.className = 'modal-overlay';
    modal.innerHTML = `

            <div class="modal-header">

                    </div>
                    <div class="weather-detail-item">

                            <span class="weather-detail-label">极端天气</span>
                            <span class="weather-detail-value" id="modal-weather-extreme">--</span>
                        </div>
                    </div>
                </div>
                <div class="weather-action-buttons">
                    <button class="modal-btn primary" onclick="openWeatherAdjustModal()">手动修改天气</button>
                    <button class="modal-btn secondary" onclick="this.closest('.modal-overlay').remove()">关闭</button>
                </div>
            </div>
        </div>
    `;
    document.body.appendChild(modal);

    // 更新弹窗数据
    document.getElementById('modal-weather-temp').textContent =
        document.getElementById('weather-temp').textContent;
    document.getElementById('modal-weather-precip').textContent =
        document.getElementById('weather-precipitation').textContent;
    document.getElementById('modal-weather-wind').textContent =
        document.getElementById('weather-wind').textContent;
    document.getElementById('modal-weather-extreme').textContent =
        document.getElementById('weather-extreme').textContent;
}
